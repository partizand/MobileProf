Функция ПолучитьНовоеЧисло() Экспорт
	
	Флаг = Истина;
	
	Пока Флаг Цикл
		УИД = Новый УникальныйИдентификатор();
	
		Запись = РегистрыСведений.УникальностьГенератораСлучайныхЧисел.СоздатьМенеджерЗаписи();
		Запись.Значение = УИД;
		Запись.Прочитать();
		Если НЕ Запись.Выбран() Тогда
			Флаг = Ложь;
			Запись.Значение = УИД;
			Запись.Записать(Истина);
		КонецЕсли;
	КонецЦикла;
	
	Стр = СтрЗаменить(Строка(УИД), "-", "");
	
	СтрСимволы = "0123456789abcdef";
	
	Кол = СтрДлина(Стр);

	н = Кол;
	к = 0;
	
	Рез = 0;
	
	Пока н >= 1 Цикл
		ТекСимвол = Символ(КодСимвола(Стр, н));
		
		Поз = Найти(СтрСимволы, ТекСимвол);
		Если Поз = 0 Тогда
			Прервать;
		КонецЕсли;
		
		Поз = Поз - 1;
		
		Рез = Рез + Поз * Pow(16, к);
		
		н = н - 1;
		к = к + 1;
	КонецЦикла;
	
	Возврат Рез;
